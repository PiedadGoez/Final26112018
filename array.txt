CREATE TYPE number_array AS VARRAY(50) OF INTEGER;
CREATE OR REPLACE FUNCTION divisible_sum_pairs(miarreglo number_array, numero number)
   RETURN number IS   
   --VARIABLES 
  modulo number; 
  esDivisible number;
  suma number;
  
  BEGIN 

    FOR i IN 1..miarreglo.count LOOP  
      FOR j IN 1..miarreglo.count+1 LOOP 
        suma:=(miarreglo(i) + miarreglo(j));
        modulo:= mod(suma, numero);
        IF modulo = 0 and i < j THEN
        esDivisible :=1;
        ELSE
        esDivisible :=0;  
        END IF;
        dbms_output.put_line('EL VALOR CALCUALDO ES  '||suma); 
      END LOOP;            
    END LOOP;     
    
   RETURN  suma;
   
  END divisible_sum_pairs;
    
    
    
    declare    
    a number_array := number_array(1, 3, 2, 6, 1, 2);
    suma number;
    begin    
     suma :=   divisible_sum_pairs(a, 3);
    
    end;
